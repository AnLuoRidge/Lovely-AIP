function TabContainer({ children, dir }) {
    return (
    <Typography component="div" dir={dir} style={{color:'white', marginTop:'7px'}}>
      {children}
    </Typography>
  );
}

TabContainer.propTypes = {
  children: PropTypes.node.isRequired,
  dir: PropTypes.string.isRequired,
};

const styles = theme => ({
